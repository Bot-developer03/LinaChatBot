```console
 git clone https://github.com/TeamYukki/YukkiChatBot
shikhar@MacBook~ $ cd YukkiChatBot
shikhar@MacBook~ $ pip3 install -r requirements.txt
```

```console
shikhar@MacBook~ $ python3 main.py
```
